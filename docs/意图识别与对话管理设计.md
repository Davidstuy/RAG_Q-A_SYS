### 2. 意图识别器
```java
@Component
@RequiredArgsConstructor
public class IntentRecognizer {
    
    private final ChatLanguageModel chatModel;
    
    public IntentResult recognize(String userInput, String context) {
        // 构建提示词
        String prompt = buildIntentPrompt(userInput, context);
        
        // 调用 LLM 进行意图识别
        String response = chatModel.generate(prompt);
        
        // 解析响应
        return parseIntentResponse(response);
    }
    
    private String buildIntentPrompt(String userInput, String context) {
        return String.format("""
            你是一个意图识别助手。请分析用户的输入，识别其意图。
            
            用户输入: %s
            上下文: %s
            
            可意图类型:
            - WEATHER_QUERY: 查询天气相关信息
            - CURRENCY_CONVERT: 货币汇率转换
            - DOCUMENT_SEARCH: 在文档中搜索信息
            - CALCULATION: 数学计算
            - REMINDER: 设置提醒
            - UNKNOWN: 无法识别的意图
            
            请返回 JSON 格式的结果:
            {
                "intent": "意图类型",
                "confidence": 0.95,
                "parameters": {
                    "city": "北京",
                    "currency": "USD"
                },
                "explanation": "识别原因"
            }
            """, userInput, context);
    }
    
    private IntentResult parseIntentResponse(String response) {
        // 解析 JSON 响应
        ObjectMapper mapper = new ObjectMapper();
        try {
            JsonNode rootNode = mapper.readTree(response);
            String intent = rootNode.get("intent").asText();
            double confidence = rootNode.get("confidence").asDouble();
            JsonNode paramsNode = rootNode.get("parameters");
            Map<String, Object> parameters = mapper.convertValue(paramsNode, new TypeReference<Map<String, Object>>() {});
            
            return new IntentResult(IntentType.valueOf(intent), confidence, parameters);
        } catch (Exception e) {
            return new IntentResult(IntentType.UNKNOWN, 0.0, Collections.emptyMap());
        }
    }
}
```

## 💬 对话管理系统

### 1. 对话上下文管理
```java
@Component
public class DialogueManager {
    
    private final Map<String, DialogueContext> contexts = new ConcurrentHashMap<>();
    
    public DialogueContext getContext(String sessionId) {
        return contexts.computeIfAbsent(sessionId, id -> new DialogueContext(id));
    }
    
    public void updateContext(String sessionId, String userInput, String response) {
        DialogueContext context = getContext(sessionId);
        context.addTurn(userInput, response);
        context.updateLastActivity();
    }
    
    public void clearContext(String sessionId) {
        contexts.remove(sessionId);
    }
}

@Data
public class DialogueContext {
    private String sessionId;
    private List<DialogueTurn> turns = new ArrayList<>();
    private LocalDateTime lastActivity;
    
    public DialogueContext(String sessionId) {
        this.sessionId = sessionId;
        this.lastActivity = LocalDateTime.now();
    }
    
    public void addTurn(String userInput, String response) {
        turns.add(new DialogueTurn(userInput, response, LocalDateTime.now()));
        // 保持最近10轮对话
        if (turns.size() > 10) {
            turns = turns.subList(turns.size() - 10, turns.size());
        }
    }
    
    public String getContextText() {
        return turns.stream()
            .map(turn -> "用户: " + turn.getUserInput() + "\n助手: " + turn.getResponse())
            .collect(Collectors.joining("\n\n"));
    }
    
    @Data
    public static class DialogueTurn {
        private String userInput;
        private String response;
        private LocalDateTime timestamp;
        
        public DialogueTurn(String userInput, String response, LocalDateTime timestamp) {
            this.userInput = userInput;
            this.response = response;
            this.timestamp = timestamp;
        }
    }
}
```

## 🎯 完整的 Agent 流程

### 1. Agent 主控制器
```java
@RestController
@RequestMapping("/api/agent")
@RequiredArgsConstructor
public class AgentController {
    
    private final IntentRecognizer intentRecognizer;
    private final ToolRegistry toolRegistry;
    private final DialogueManager dialogueManager;
    private final DialogueStateManager dialogueStateManager;
    
    @PostMapping("/chat")
    public ResponseEntity<?> chat(@RequestBody AgentRequest request) {
        try {
            String sessionId = request.getSessionId();
            String userInput = request.getMessage();
            
            // 1. 获取对话上下文
            DialogueContext context = dialogueManager.getContext(sessionId);
            
            // 2. 意图识别
            IntentResult intent = intentRecognizer.recognize(userInput, context.getContextText());
            
            // 3. 检查参数完整性
            if (!intent.isConfident() || !checkParametersComplete(intent)) {
                // 请求补充参数
                return ResponseEntity.ok(AgentResponse.parameterNeeded(intent));
            }
            
            // 4. 选择工具
            Tool tool = toolRegistry.getTool(getToolNameForIntent(intent.getIntent()));
            if (tool == null) {
                return ResponseEntity.ok(AgentResponse.error("无法找到合适的工具"));
            }
            
            // 5. 执行工具
            ToolInput input = createToolInput(intent, sessionId);
            ToolExecution result = tool.execute(input);
            
            // 6. 更新对话上下文
            dialogueManager.updateContext(sessionId, userInput, result.getResult().toString());
            
            // 7. 返回结果
            return ResponseEntity.ok(AgentResponse.success(result.getResult()));
            
        } catch (Exception e) {
            return ResponseEntity.internalServerError().body(AgentResponse.error(e.getMessage()));
        }
    }
    
    private boolean checkParametersComplete(IntentResult intent) {
        // 根据不同意图检查参数完整性
        switch (intent.getIntent()) {
            case WEATHER_QUERY:
                return intent.getParameters().containsKey("city");
            case CURRENCY_CONVERT:
                return intent.getParameters().containsKey("from") && 
                       intent.getParameters().containsKey("to") && 
                       intent.getParameters().containsKey("amount");
            default:
                return true;
        }
    }
    
    private String getToolNameForIntent(IntentType intent) {
        switch (intent) {
            case WEATHER_QUERY:
                return "weather_query";
            case CURRENCY_CONVERT:
                return "currency_converter";
            case DOCUMENT_SEARCH:
                return "document_search";
            default:
                return "document_search"; // 默认使用文档搜索
        }
    }
}
```

### 2. 请求和响应模型
```java
@Data
public class AgentRequest {
    private String sessionId;
    private String message;
    private Map<String, Object> context;
}

@Data
public class AgentResponse {
    private boolean success;
    private String message;
    private Object data;
    private AgentResponseType type;
    
    public static AgentResponse success(Object data) {
        AgentResponse response = new AgentResponse();
        response.setSuccess(true);
        response.setData(data);
        response.setType(AgentResponseType.SUCCESS);
        return response;
    }
    
    public static AgentResponse parameterNeeded(IntentResult intent) {
        AgentResponse response = new AgentResponse();
        response.setSuccess(false);
        response.setMessage("请提供更多信息: " + getMissingParameters(intent));
        response.setType(AgentResponseType.PARAMETER_NEEDED);
        return response;
    }
    
    public static AgentResponse error(String message) {
        AgentResponse response = new AgentResponse();
        response.setSuccess(false);
        response.setMessage(message);
        response.setType(AgentResponseType.ERROR);
        return response;
    }
}

public enum AgentResponseType {
    SUCCESS,
    PARAMETER_NEEDED,
    ERROR
}
```
